import { Tabs, TabItem } from '@astrojs/starlight/components';

:::note
Catchphrase can be installed by using `npm`, `Yarn`, `pnpm` or `Bun`.
:::

## React

<Tabs>
<TabItem value="npm" label="npm">

```bash
npm i @catchphrase/react
```

</TabItem>
<TabItem value="yarn" label="Yarn">

```bash
yarn add @catchphrase/react
```

</TabItem>
<TabItem value="pnpm" label="pnpm">

```bash
pnpm i @catchphrase/react
```

</TabItem>
<TabItem value="bun" label="Bun">

```bash
bun add @catchphrase/react
```

</TabItem>
</Tabs>

:::note
There will be a Catchphrase CLI in the future that will make it easier to install Catchphrase.
:::
